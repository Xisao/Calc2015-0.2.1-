import java.util.Scanner; public class calc { public static void main(String[] args) { Scanner scanner = new Scanner(System.in); { System.out.println("Введите нужную операцию например 6 - 3"); double n1 = scanner.nextDouble(); String operation = scanner.next(); double n2 = scanner.nextDouble(); switch (operation) { case "+": System.out.println("Думаю ваш ответ это " + (n1 + n2)); break; case "-": System.out.println("Думаю ваш ответ это " + (n1 - n2)); break; case "/": System.out.println("Думаю ваш ответ это " + (n1 / n2)); break; case "*": System.out.println("Думаю ваш ответ это " + (n1 * n2)); break; default: System.out.println("Ошибка не могу сосчитать, отключаюсь!"); } } } }
